# Palindrome

n = int(input("Enter the number "))

rev = 0
num = n

while num > 0:
    rem = num % 10
    num = int(num / 10)
    rev = rev * 10 + rem

if n == rev:
    print(n, "is a palindrome")
else:
    print(n, "is not a palindrome")

# Swapping using function

def swap(num1, num2):
    if num1 < num2:
        return num2, num1
    else:
        return num1, num2

number1 = int(input("Enter the value for number1"))
number2 = int(input("Enter the value for number2"))

result1, result2 = swap(number1, number2)

print("Original numbers: number1=", number1, "number2=", number2)
print("After swapping: number1=", result1, "number2=", result2)
